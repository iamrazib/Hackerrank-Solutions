class MyCalculator {
    
    public static long power(int n,int p) throws Exception
    {
        long ln = n;
        long lp = p;
        
        long c = (long)Math.pow(ln,lp);
        
        if(ln==0 && lp==0){
            throw new Exception("n and p should not be zero.");
        }
        else if(ln<0 || lp<0){
            throw new Exception("n or p should not be negative.");
        }         
        else{
            return c;
        }
    }
}
